CREATE DATABASE SCAS;

CREATE TABLE SCAS.CURSO(
  CURSO_ID INT NOT NULL,
  NOME VARCHAR(45) NULL,
  TIPO_ENSINO VARCHAR(45) NULL,
  TURNO VARCHAR(45) NULL,
  PRIMARY KEY (CURSO_ID)
);

CREATE TABLE SCAS.USUARIO (
  USUARIO_ID INT NOT NULL,
  DATA_NASC VARCHAR(45) NULL,
  NOME VARCHAR(45) NULL,
  SEXO VARCHAR(45) NULL,
  CPF VARCHAR(14) NULL,
  IDENTIDADE VARCHAR(45) NULL,
  TELEFONE_FIXO VARCHAR(15) NULL,
  TELEFONE_CELULAR VARCHAR(15) NULL,
  EMAIL VARCHAR(45) NULL,
  ENDERECO VARCHAR(45) NULL,
  NUMERO VARCHAR(45) NULL,
  COMPLEMENTO VARCHAR(45) NULL,
  BAIRRO VARCHAR(45) NULL,
  CEP VARCHAR(45) NULL,
  CIDADE VARCHAR(45) NULL,
  UF VARCHAR(45) NULL,
  LOGIN VARCHAR(45) NULL,
  SENHA VARCHAR(45) NULL,
  PRIMARY KEY (USUARIO_ID)
);


CREATE TABLE SCAS.FUNCIONARIO(
  REGISTRO INT NOT NULL,
  USUARIO_ID INT NOT NULL,
  CARGO VARCHAR(45) NULL,
  PRIMARY KEY (REGISTRO),
    FOREIGN KEY (USUARIO_ID) REFERENCES SCAS.USUARIO (USUARIO_ID)
);

CREATE TABLE SCAS.ALUNO(
  MATRICULA INT NOT NULL,
  ANO_INGRESSO INT NULL,
  PERIODO_CURSO VARCHAR(45) NULL,
  FAMILIA_ENDERECO VARCHAR(45) NULL,
  FAMILIA_NUMERO VARCHAR(45) NULL,
  FAMILIA_COMPLEMENTO VARCHAR(45) NULL,
  FAMILIA_BAIRRO VARCHAR(45) NULL,
  FAMILIA_CEP VARCHAR(45) NULL,
  FAMILIA_CIDADE VARCHAR(45) NULL,
  FAMILIA_UF VARCHAR(45) NULL,
  CURSO_ID INT NOT NULL,
  USUARIO_ID INT NOT NULL,
  PRIMARY KEY (MATRICULA),
    FOREIGN KEY (CURSO_ID) REFERENCES SCAS.CURSO (CURSO_ID),
    FOREIGN KEY (USUARIO_ID) REFERENCES SCAS.USUARIO (USUARIO_ID)
);

CREATE TABLE SCAS.MODALIDADE(
  MODALIDADE_ID INT NOT NULL,
  VALOR_MENSAL DOUBLE NULL,
  NOME VARCHAR(45) NULL,
  DESCRICAO VARCHAR(45) NULL,
  PRIMARY KEY (MODALIDADE_ID)
);

CREATE TABLE SCAS.RECURSO(
  RECURSO_ID INT NOT NULL,
  ANO INT NULL,
  CREDITOS DOUBLE NULL,
  DEBITOS DOUBLE NULL,
  SALDO DOUBLE NULL,
  MODALIDADE_ID INT NOT NULL,
  PRIMARY KEY (RECURSO_ID),  
    FOREIGN KEY (MODALIDADE_ID) REFERENCES SCAS.MODALIDADE (MODALIDADE_ID)
);

CREATE TABLE SCAS.SELECAO(
  SELECAO_ID INT NOT NULL,
  DT_INICIO_INSCRICAO VARCHAR(10) NULL,
  DT_FIM_INSCRICAO VARCHAR(10) NULL,
  NUM_EDITAL VARCHAR(45) NULL,
  MODALIDADE_ID INT NOT NULL,
  PRIMARY KEY (SELECAO_ID),
    FOREIGN KEY (MODALIDADE_ID) REFERENCES SCAS.MODALIDADE (MODALIDADE_ID)
);

CREATE TABLE SCAS.BOLSA(

CREATE TABLE SCAS.DOENCA(
  DOENCA_ID INT NOT NULL,

  QT17_NOME VARCHAR(45) NULL,
